# SegmentedButton

Button group component to select one or multiple options. Each unit Button just extends the Button component.
Supports icons, multi-select, custom modes, sizes, statuses, and full styling customization.

import {
  BlossomUIRenderer,
  PropsTable,
  SegmentedButtonUsage,
  SegmentedButtonMultiSelect,
  SegmentedButtonIcons,
  SegmentedButtonModes,
  SegmentedButtonSizes,
  SegmentedButtonStatuses,
  SegmentedButtonDisabled,
  SegmentedButtonCustom,
} from "@site/src/components";

Following values will be used on the following demos:

```tsx
const buttons: ControlButtonProps[] = [
  { title: "Primary" },
  { title: "Secondary" },
  { title: "Tertiary" },
];

const iconButtons: ControlButtonProps[] = [
  { title: "Primary", left: <Icon name="add" /> },
  { title: "Secondary", left: <Icon name="add" /> },
  { title: "Tertiary", left: <Icon name="add" /> },
];
```
