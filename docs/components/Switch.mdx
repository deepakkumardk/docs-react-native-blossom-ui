# Switch

Custom switch component with optional label, caption, and error support.  
Supports different positions, sizes, statuses, default values, and full styling.

import {
  BlossomUIRenderer,
  PropsTable,
  SwitchUsage,
  SwitchPositions,
  SwitchStatuses,
  SwitchSizes,
  SwitchDisabled,
  SwitchDefaultValue,
  SwitchCustom,
} from "@site/src/components";

## Usage

<BlossomUIRenderer
  componentName={"SwitchUsage"}
  children={<SwitchUsage />}
  fileName={"Switch"}
/>
