# View

The most basic container component, extending React Native's `View`.  
Automatically applies the current theme background color.

import {
  BlossomUIRenderer,
  PropsTable,
  ViewUsage,
  ViewRow,
} from "@site/src/components";

## Usage

<BlossomUIRenderer
  componentName={"ViewUsage"}
  children={<ViewUsage />}
  fileName={"View"}
/>

## Row

Use the `row` prop or style to layout children horizontally.

<BlossomUIRenderer
  componentName={"ViewRow"}
  children={<ViewRow />}
  fileName={"View"}
/>
